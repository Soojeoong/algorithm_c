5.대각선 배열
N x M (1≤N, M≤100) 크기의 행렬에 1 ~ MN 의 수를 아래 그림과 같이 ↙ 대각선 방향으로 채운 결과를 출력하시오.

(행과 열 번호의 합이 같다)
if (i+j == 0,1,2,3,4...)

```c
int main() {
	int x[100][100], k, N, M;
	int num = 1, num_max = 0;
	int i, j;

	scanf("%d %d", &N, &M); //행렬 크기 N, M
	num_max = N * M; //총 숫자

 	for (k = 0; k < num_max; k++) { //행렬 크기만큼
		for (i = 0; i < N; i++) { //행은 계속++
			j = k - i; //j는 0, (1,0) , (2,1,0), (3,2,1,0)식
			if (j >= 0 && j < M) {
				x[i][j] = num; //채우기
				num++; //x[i][j] = num++;
			}
		}
	}

	for (i = 0; i < N; i++) //출력
	{
		for (j = 0; j < M; j++)
		{
			printf(" %d", x[i][j]);
		}
		printf("\n");
	}

	return 0;
}
```
