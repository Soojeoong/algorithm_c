3. 위에서부터 -->, <-- 번갈아 채우기

N x N (1≤N≤100) 크기의 행렬에 1 ~ N² 의 수를 아래 그림과 같이 차례로 위에서부터 → 방향과 ← 방향을 번갈아 가면서 채운 결과를 출력하시오

```c
int main() {
	int x[100][100]; //이차원 배열
	int N, i, j; //N : 행렬 크기
	int cnt = 0; //출력 숫자

	scanf("%d", &N); //행렬 크기 N 입력

	for (i = 0; i < N; i++) {
		if (i % 2 == 0) {//짝수 행
			for (j = 0; j < N; j++) {
				x[i][j] = ++cnt; //--> 순으로 숫자 넣기
			}
		}
		else { //홀수 행
			for (j = N - 1; j >= 0; j--) { 
				x[i][j] = ++cnt; //<-- 순으로 숫자 넣기
			}
		}
	}

	//N*N 행렬 출력
	for (i = 0; i < N; i++) {	
		for (j = 0; j < N; j++) {
			printf(" %d", x[i][j]);
		}
		printf("\n");
	}

	return 0;
 }
 ```
